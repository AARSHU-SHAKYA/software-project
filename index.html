Write a C program to write a character and display it using fgetc() and fputc() functions.

#include<stdio.h>
int main(){
    FILE *fp;
    char ch, c;
    fp = fopen("./filechar.txt", "w+");
    printf("Enter a character: ");
    scanf("%s", &ch);
    fputc(ch, fp);
    printf("Reading the file\n");
    rewind(fp);
    c = fgetc(fp);
    printf("The entered character is %c", c);
